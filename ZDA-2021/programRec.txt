int function fac(int i) 
{
   where (i ~ 0) {
      ret 1;
   }
   else {
      let int z = i - 1;
      let int res = i * fac(z);
      ret res;
   }   
}

main
{
   let int res = fac(0x6);
   write 'Результат = ';

}
